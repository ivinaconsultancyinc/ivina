from flask import flash @app.route('/login', methods=['GET', 'POST']) def login(): if request.method == 'POST': username = request.form['username'] password = request.form['password'] user = User.query.filter_by(username=username).first() if user and check_password_hash(user.password, password): session['user_id'] = user.id return redirect(url_for('dashboard')) flash('Invalid username or password') return redirect(url_for('login')) return render_template('login.html')
